#include <rtthread.h>
#include <rtdevice.h>
#include <drv_common.h>

#define BEEF_PIN GET_PIN(A,5)
#define KEY1_PIN GET_PIN(A,0)

int main(void)
{
    rt_pin_mode(BEEF_PIN, PIN_MODE_OUTPUT);
    rt_pin_mode(KEY1_PIN, PIN_MODE_INPUT_PULLUP);

    while (1)
    {
        if(PIN_LOW==rt_pin_read(KEY1_PIN))
        {
            rt_thread_mdelay(20);
            if(PIN_LOW==rt_pin_read(KEY1_PIN))
            {
                rt_pin_write(BEEF_PIN, PIN_LOW);
            }
        }
        else {
            rt_pin_write(BEEF_PIN, PIN_HIGH);
        }
        rt_thread_mdelay(300);

    }

    return RT_EOK;
}
